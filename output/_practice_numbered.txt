  150:   function extractOcrInfo(ocrResult) {
  151:     const info = { text: '', letter: '', confidence: 0 };
  152:     if (!ocrResult || !ocrResult.data) return info;
  153:     const data = ocrResult.data;
  154:     info.text = (data.text || '').trim();

  156:     function pushLetter(ch, conf) {
  157:       if (!ch) return;
  158:       const l = String(ch).toLowerCase();
  159:       if (!/^[a-z]$/.test(l)) return;
  160:       const c = (typeof conf === 'number' && isFinite(conf)) ? conf : 0;
  161:       if (!info.letter || c > info.confidence) {
  162:         info.letter = l;
  163:         info.confidence = c;
  164:       }
  165:     }
  166: 
  167:     if (Array.isArray(data.symbols) && data.symbols.length > 0) {
  168:       data.symbols.forEach(sym => {
  169:         const ch = sym.text || sym.symbol || '';
  170:         pushLetter(ch, sym.confidence);
  171:       });
  172:     } else if (Array.isArray(data.words) && data.words.length > 0) {
  173:       data.words.forEach(word => {
  174:         const text = word.text || '';
  175:         const conf = word.confidence;
  176:         for (let i = 0; i < text.length; i++) {
  177:           pushLetter(text[i], conf);
  178:         }
  179:       });
  180:     }
  181: 
  182:     if (!info.letter && info.text) {
  183:       const lettersOnly = info.text.toLowerCase().replace(/[^a-z]/g, '');
  184:       if (lettersOnly.length === 1) {
  185:         info.letter = lettersOnly[0];
  186:         const avgConf = (typeof data.confidence === 'number' && isFinite(data.confidence)) ? data.confidence : 0;
  187:         info.confidence = avgConf;
  188:       }
  189:     }
  190: 
  191:     return info;
  192:   }

  298:   function buildStudyXpPayload(bestMap, source) {
  299:     const list = getStudyKanaList();
  300:     const categoryLabelMap = {
  301:       basic: '\u6e05\u97f3',
  302:       dakuten: '\u6fc1\u97f3\u30fb\u534a\u6fc1\u97f3',
  303:       yoon: '\u62d7\u97f3',
  304:       other: '\u305d\u306e\u4ed6'
  305:     };
  306: 
  307:     const categories = {};
  308:     Object.keys(categoryLabelMap).forEach(function (cat) {
  309:       categories[categoryLabelMap[cat]] = { count80: 0, count90: 0 };
  310:     });
  311: 
  312:     let count80 = 0;
  313:     let count90 = 0;
  314:     list.forEach(function (item) {
  315:       const score = bestMap && typeof bestMap[item.kana] === 'number' ? bestMap[item.kana] : 0;
  316:       const key = categoryLabelMap[item.category] || categoryLabelMap.other;
  317:       if (score >= 80) {
  318:         count80 += 1;
  319:         categories[key].count80 += 1;
  320:       }
  321:       if (score >= 90) {
  322:         count90 += 1;
  323:         categories[key].count90 += 1;
  324:       }
  325:     });
  326: 
  327:     const rawStudyXP = (count80 + count90) / 4;
  328:     // Rounding policy: keep one decimal place (Math.round(value * 10) / 10).
  329:     const studyXP = Math.round(rawStudyXP * 10) / 10;
  330: 
  331:     return {
  332:       timestamp: new Date().toISOString(),
  333:       totalKana: list.length,
  334:       count80: count80,
  335:       count90: count90,
  336:       studyXP: studyXP,
  337:       categories: categories,
  338:       source: source
  339:     };
  340:   }
